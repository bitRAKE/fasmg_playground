macro JRCXNZ? target*
	inc rcx
	loop target
end macro



; testing if a register is zero

jrcxz target

test reg,reg
jz target



;testing if a register is one

dec reg
jz target


;testing if a register is -1

inc reg
jz target


; reg is NOT boolean value (0 or 1)

test reg,-2	; shr reg,1
jnz target
